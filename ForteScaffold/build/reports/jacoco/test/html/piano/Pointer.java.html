package piano;

import processing.core.PApplet;
import processing.core.PImage;
//import javax.sound.midi.*;

public class Pointer {

    private int x;
    private int y;

    private boolean active;
    private int vel;
    //private boolean pianoactive;

    private PImage sprite;

    public Pointer(int x,int y,PImage sprite){
        this.x = x;
        this.y = y;

        this.active = false;
        this.vel = 1;
        //this.pianoactive = false;

        this.sprite = sprite;
    }

    public void tick(PianoRoll piano){
        //handles logic
        

        if(this.active){
            this.x+=this.vel;
            
            if(this.x > 528){
                this.x = 48;
            }
        }

    }

    public void draw(PApplet app){
        //handles graphics
        app.image(this.sprite,this.x,this.y);
        app.line(this.x+12,75,this.x+12,355);
        app.strokeWeight(1);
        //app.colorMode(255);
        app.stroke(255,0,0);
    }

    public void activate(){
        this.active = !this.active;
        
    }

    public void reset(){
        this.x = 48;
        this.active = false;
    }

    public int returnX(){
        return this.x;
    }

}
